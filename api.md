# 轮播图管理

#### 请求地址
    /api/banner
#### 请求方式
    GET
#### 请求参数
    {
      id:number, [number：必填]
      name: "",  [字符串：必填]
      imageUrl: "路径",  [字符串：必填]
      start: 0, //记录开始值 [数值：必填]
      count: 4 //返回记录条数 [数值：必填]
    }
#### 返回数据
	{
      "errcode": 0, // 错误代码：[数值：必填] 0 无错误 -1 有错误
      "errmsg": "", // 错误信息：[字符串：默认为空]
      "data": { // 数据内容
        start: 0, //记录开始值 [数值：必填]
        count: 4, //返回记录条数 [数值：必填]
        total: 10, //总记录条数 [数值：必填]
        subjects: [
          {
            id:0,  [number：必填]
            name: '环游客1', // banner图名字 [字符串：必填]
            imageUrl: 'http://....',图片路径 [字符串：必填]
          },
          ......
        ]
      }
	}
